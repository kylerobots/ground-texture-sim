41 def _write_array(self, array: numpy.ndarray, file_path: str) -> None:
42 """!
43 @brief Write a Numpy array to the given file.
44
45 This formats it as one row per line, with spaces as delimiters. Numbers are formatted as
46 decimal numbers with 6 places after the decimal.
47
48 @param array The array to write to file. This must be a 2D array.
49 @param file_path The file to write to. The path can be absolute or relative. The containing
50 folder must exist.
51 @return None
52 """
53 # Create a nicely formatted string first.
54 output = ''
55 for i in range(array.shape[0]):
56 for j in range(array.shape[1]):
57 output += F'{array[i, j]:0.6f} '
58 # At the end of the row, remove the trailing space and add a newline.
59 output = output.strip()
60 output += '\n'
61 # Now write to file
62 with open(file=file_path, mode='w', encoding='utf-8') as file:
63 file.write(output)

101 def write_lists(self, robot_poses: List[List[float]], pixel_poses: List[List[float]]) -> None:
102 """!
103 @brief Write all the data points in the provided lists to files names according to the data
104 format standard.
105
106 The image names are automatically determined, since they follow a consistent format. It is
107 assumed that the first element in each list corresponds to the first image, etc.
108
109 @param robot_poses A list of ground truth robot poses, in the form [x, y, yaw]. X and Y are
110 in meters and yaw is in radians.
111 @param pixel_poses A list of poses for the top left corner of the image, relative to a
112 global image aligned with the origin. Each element is in the form [x, y, yaw] where X and Y
113 are in pixels and yaw is in radians.
114 @return None
115 @exception ValueError returned if both lists are not identical in length.
116 """
117 # Ensure each list is the same size, otherwise, the alternating lists will be screwed up.
118 if len(robot_poses) != len(pixel_poses):
119 raise ValueError('Provided lists must be the same length.')
120 # Derive the list of image paths, making sure a newline will get written.
121 image_paths = []
122 for i in range(len(robot_poses)):
123 image_paths.append(
124 self._namer_namer.create_image_path(i, absolute=False) + '\n')
125 # Now write each file in turn. Start with the test file, which is just images.
126 test_file_path = os.path.join(
127 self._output_directory_output_directory, self._namer_namer.test_file)
128 with open(file=test_file_path, mode='w', encoding='utf-8') as test_file:
129 for image_path in image_paths:
130 test_file.write(image_path)
131 # For the next two, alternate with the appropriate pose. Convert the yaw into a matrix.
132 meters_txt_file_path = os.path.join(
133 self._output_directory_output_directory, self._namer_namer.meters_txt_file)
134 with open(file=meters_txt_file_path, mode='w', encoding='utf-8') as meters_txt_file:
135 for image_path, robot_pose in zip(image_paths, robot_poses):
136 meters_txt_file.write(image_path)
137 # Since there is only a yaw, we can safely extract the upper part of the matrix
138 rotation = create_transform_matrix(
139 robot_pose[0], robot_pose[1], 0.0, 0.0, 0.0, robot_pose[2])
140 pose_string = F'{rotation[0, 0]:0.6f} {rotation[0, 1]:0.6f} {rotation[0, 3]:0.6f}' \
141 F' {rotation[1, 0]:0.6f} {rotation[1, 1]:0.6f} {rotation[1, 3]:0.6f} ' \
142 F'{0:0.6f} {0:0.6f} {1:0.6f}\n'
143 meters_txt_file.write(pose_string)
144 txt_file_path = os.path.join(
145 self._output_directory_output_directory, self._namer_namer.txt_file)
146 with open(file=txt_file_path, mode='w', encoding='utf-8') as txt_file:
147 for image_path, pixel_pose in zip(image_paths, pixel_poses):
148 txt_file.write(image_path)
149 # Since there is only a yaw, we can safely extract the upper part of the matrix
150 rotation = create_transform_matrix(
151 pixel_pose[0], pixel_pose[1], 0.0, 0.0, 0.0, pixel_pose[2])
152 pose_string = F'{rotation[0, 0]:0.6f} {rotation[0, 1]:0.6f} {rotation[0, 3]:0.6f}' \
153 F' {rotation[1, 0]:0.6f} {rotation[1, 1]:0.6f} {rotation[1, 3]:0.6f} ' \
154 F'{0:0.6f} {0:0.6f} {1:0.6f}\n'
155 txt_file.write(pose_string)
