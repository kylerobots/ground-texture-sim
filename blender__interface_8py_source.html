30 This is a 3x3 matrix derived from Blender using these steps:
31 https://visp-doc.inria.fr/doxygen/visp-3.4.0/tutorial-tracking-mb-generic-rgbd-Blender.html
32
33 @return A 3x3 numpy array representing the intrinsic matrix.
34 """
35 camera = bpy.data.objects[self.camera_namecamera_namecamera_namecamera_name]
36 focal_length = camera.data.lens
37 scene = bpy.context.scene
38 resolution_x_px = scene.render.resolution_x
39 resolution_y_px = scene.render.resolution_y
40 scale = scene.render.resolution_percentage / 100.0
41 sensor_width_mm = camera.data.sensor_width
42 sensor_height_mm = camera.data.sensor_height
43 aspect_ratio = scene.render.pixel_aspect_x / scene.render.pixel_aspect_y
44 if camera.data.sensor_fit == 'VERTICAL':
45 s_u = resolution_x_px * scale / sensor_width_mm / aspect_ratio
46 s_v = resolution_y_px * scale / sensor_height_mm
47 else:
48 aspect_ratio = scene.render.pixel_aspect_x / scene.render.pixel_aspect_y
49 s_u = resolution_x_px * scale / sensor_width_mm
50 s_v = resolution_y_px * scale * aspect_ratio / sensor_height_mm
51 alpha_u = focal_length * s_u
52 alpha_v = focal_length * s_v
53 u_0 = resolution_x_px * scale / 2
54 v_0 = resolution_y_px * scale / 2
55 skew = 0
56 matrix = numpy.array([
57 [alpha_u, skew, u_0],
58 [0.0, alpha_v, v_0],
59 [0.0, 0.0, 1.0]

88 def generate_image(self, image_path: str, camera_pose: numpy.ndarray) -> None:
89 """!
90 @brief Position the camera at a designated pose and render an image.
91
92 This method takes the 6 DOF pose of the camera, places the camera at that pose in the
93 environment, renders the image, then saves the image to the designated file.
94
95 Blender's coordinate system is different than the usual coordinate system assumed in
96 robotics. This method adds the correct adjustment, so there is no need to account for it. In
97 other words, specify the camera pose using conventional robotics coordinate systems.
98
99 @param image_path An absolute path to where the image should go. Blender does not seem to
100 handle relative paths great.
101 @param camera_pose The 4x4 homogenous matrix representing the pose of the camera in the
102 world frame.
103 @return None
104 @exception ValueError raised if the provided path is not absolute.
105 """
106 # Verify the path is absolute.
107 if not path.isabs(image_path):
108 raise ValueError(
109 F'Image path must be absolute. Received: {image_path}')
110 # Convert the provided pose into a Mathutils matrix. This has better support for extracting
111 # the XYZ, RPY values needed to position in Blender.
112 pose = mathutils.Matrix()
113 for i in range(4):
114 for j in range(4):
115 pose[i][j] = camera_pose[i][j]
116 # Calculate the offset to get Blender to position the camera correctly.
117 # Pre-compute the offset used to get blender camera poses correct.
118 blender_adjustment = mathutils.Euler(
119 (pi/2.0, 0.0, -pi/2.0), 'XYZ').to_matrix().to_4x4()
120 blender_placement = pose @ blender_adjustment
121 # Now place the camera
122 camera = bpy.data.objects[self.camera_namecamera_namecamera_namecamera_name]
123 camera.location.x = blender_placement.to_translation()[0]
124 camera.location.y = blender_placement.to_translation()[1]
125 camera.location.z = blender_placement.to_translation()[2]
126 camera.rotation_euler.x = blender_placement.to_euler('XYZ')[0]
127 camera.rotation_euler.y = blender_placement.to_euler('XYZ')[1]
128 camera.rotation_euler.z = blender_placement.to_euler('XYZ')[2]
129 # Render the image
130 bpy.context.scene.render.filepath = image_path
131 bpy.ops.render.render(write_still=True)
