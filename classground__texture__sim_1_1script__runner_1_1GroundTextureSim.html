53 def run(self) -> None:
54 """!
55 @brief Generate and write the data to file.
56 @return None
57 """
58 # For each trajectory, convert it into a camera pose and write the image. Capture the pixel
59 # values of the image corners too
60 pixel_poses = []
61 for i, trajectory in enumerate(self._trajectory):
62 robot_pose = ground_texture_sim.transforms.create_transform_matrix(
63 trajectory[0], trajectory[1], 0.0, 0.0, 0.0, trajectory[2])
64 # Convert to a camera pose
65 camera_pose = self._transformer.transform_camera_to_world(
66 robot_pose)
67 # Write camera image
68 image_path = self._namer.create_image_path(i, absolute=True)
69 self._blender_interface.generate_image(image_path, camera_pose)
70 # Project the image corner
71 pixel_poses.append(
72 self._transformer.project_image_corner(robot_pose))
73 # Write main list files.
74 self._writer.write_camera_intrinsic_matrix(
75 self._blender_interface.camera_intrinsic_matrix)
76 self._writer.write_camera_pose(self._camera_pose)
77 self._writer.write_lists(self._trajectory, pixel_poses)
