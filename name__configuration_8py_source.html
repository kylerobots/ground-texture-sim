39 def create_image_path(self, index: int, absolute: bool = False) -> str:
40 """!
41 @brief Assemble the directory and name of a particular image to use for its path.
42
43 From the output directory, this will go in:
44 ```
45 sequence_type/date_collected/sequence_number/
46 HDG$VersionNumber_t$TextureNumber_$SequenceType_
47 $DateOfRecording_s$SequenceNumber_c01_i$ImageNumber.png
48 ```
49
50 @param index The image number to use
51 @param absolute True if the returned path should be absolute, false if it should be relative
52 to *output*.
53 @return A relative path to the image location, relative to the output directory.
54 """
55 # Create the directory relative to the output folder.
56 date_folder = self._current_date_current_date.strftime('%y%m%d')
57 sequence_number_folder = F'seq{self._sequence_number:04d}'
58 file_directory = path.join(
59 self._sequence_type_sequence_type, date_folder, sequence_number_folder)
60 # Then the filename
61 texture_name = F't{self._texture_number:03d}'
62 date_name = self._current_date_current_date.strftime('%Y-%m-%d')
63 sequence_number_name = F's{self._sequence_number:04d}'
64 image_name = F'i{index:07d}'
65 file_name = F'HDG2_{texture_name}_{self._sequence_type}_{date_name}' \
66 F'_{sequence_number_name}_{self._camera_name}_{image_name}.png'
67 result = path.join(file_directory, file_name)
68 # Make absolute or relative, depending on the specification
69 if absolute:
70 result = path.join(self._output_folder_output_folder, result)
71 return result
